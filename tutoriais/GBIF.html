Download GBIF case files + images

In this tutorial I show a complete R script to download GBIF (DwC-A) occurrences, cross-reference them with associated media, and automatically download renamed images with metadata. You can adjust the taxon (species, genus, family, etc.) and the destination folder.

Script R: Download of Incidents and Images

# ========================================================================
# 1. LOAD PACKAGES
# ========================================================================
# install.packages(c("rgbif", "dplyr", "stringr", "purrr", "openxlsx", "finch"))

library(rgbif)     # Interface to the GBIF API – download occurrence data and metadata
library(dplyr)     # Data manipulation (filter, select, mutate, join, summarize, etc.)
library(stringr)   # String processing and text cleaning functions
library(purrr)     # Functional programming tools (loops, mapping functions)
library(openxlsx)  # Export data frames to Excel (.xlsx) files without Java dependency
library(finch)     # Read Darwin Core Archive (DWCA) files directly from GBIF (.zip)

# ========================================================================
# 2. DEFINE WORKING DIRECTORY AND TAXON
# ========================================================================

# Set main directory for Microgramma mauritiana files
setwd("C:/Users/nikso/OneDrive/Academic_and_Personal_Life_Niksoney/Articles_and_Collaborations/_Microgramma_mauritiana")

# Retrieve taxonKey from GBIF backbone taxonomy
name_backbone(name = "Microgramma mauritiana")

# Insert the taxonKey here
key <- 4026159

# ========================================================================
# 3. DOWNLOAD DWCA FROM GBIF
# ========================================================================

# Request DWCA download (required to access multimedia metadata)
dl <- occ_download(
  pred("taxonKey", key),
  format = "DWCA"
)

# Check download status (PREPARING, RUNNING, SUCCEEDED, FAILED)
occ_download_meta(dl)

# Download generated file
file <- occ_download_get(dl)

# ========================================================================
# 4. READ DWCA FILE (GBIF ZIP)
# ========================================================================

# Manually unzip downloaded file
unzip("0006490-260226173443078.zip", exdir = "gbif_dwca")

# Read occurrence.txt (Occurrence data)
occ <- read.delim(
  "gbif_dwca/occurrence.txt",
  sep = "\t",
  quote = "",
  fill = TRUE,
  stringsAsFactors = FALSE
)

# Read multimedia.txt (Media links and types)
media <- read.delim(
  "gbif_dwca/multimedia.txt",
  sep = "\t",
  quote = "",
  fill = TRUE,
  stringsAsFactors = FALSE
)

# ========================================================================
# 5. SUMMARY STATISTICS
# ========================================================================

cat("Total occurrences:", nrow(occ), "\n")
cat("Total media records:", nrow(media), "\n")
cat("Number of images (StillImage):", 
    sum(media$type == "StillImage", na.rm = TRUE), "\n")

# Check media type diversity
table(media$type)

# ========================================================================
# 6. FILTER AND PROCESS IMAGE RECORDS
# ========================================================================

# Filter image records
media_img <- media[media$type == "StillImage", ]

# Identify unique GBIF IDs with images
ids_img <- unique(media_img$gbifID)

# Add control column in occurrence table
df <- occ
df$hasImage <- ifelse(df$gbifID %in% ids_img, "YES", "NO")

# Move column to the beginning for easier inspection
df <- df %>% relocate(hasImage)

# ========================================================================
# 7. EXPORT OUTPUT FILES
# ========================================================================

write.csv(df, "Microgramma_mauritiana_GBIF.csv", row.names = FALSE)
write.xlsx(df, "Microgramma_mauritiana_GBIF.xlsx")

# ========================================================================
# 8. PREPARE IMAGE DOWNLOAD
# ========================================================================

# Merge occurrence data with image metadata
df_img <- occ %>%
  inner_join(media_img, by = "gbifID")

# Create output folder
dir.create("images", showWarnings = FALSE)

# Function to sanitize file names (Windows compatible)
sanitize_name <- function(x) {
  if (is.na(x) || x == "") return("NA")
  x <- trimws(x)
  x <- gsub("[\\/\\:*?\"<>|]", "", x)
  x <- gsub("\\s+", " ", x)
  return(x)
}

# Helper function to handle NA values
handle_na <- function(x) {
  if (is.na(x) || x == "") return("NA") else return(x)
}

# ========================================================================
# 9. CORE DOWNLOAD AND RENAME FUNCTION
# ========================================================================

download_image <- function(i) {
  
  url <- df_img$identifier[i]
  
  # Extract metadata
  sp       <- sanitize_name(df_img$species[i])
  collector<- sanitize_name(df_img$recordedBy[i])
  rec_num  <- sanitize_name(df_img$recordNumber[i])
  herb     <- sanitize_name(df_img$collectionCode[i])
  cat_num  <- sanitize_name(df_img$catalogNumber[i])
  country  <- sanitize_name(df_img$level0Name[i])
  lat      <- sanitize_name(df_img$decimalLatitude[i])
  lon      <- sanitize_name(df_img$decimalLongitude[i])
  
  # Handle missing values
  sp        <- handle_na(sp)
  collector <- handle_na(collector)
  rec_num   <- handle_na(rec_num)
  herb      <- handle_na(herb)
  cat_num   <- handle_na(cat_num)
  country   <- handle_na(country)
  lat       <- handle_na(lat)
  lon       <- handle_na(lon)
  
  # Build file name
  filename <- paste(sp, collector, rec_num, herb, cat_num,
                    country, lat, lon, sep = "_")
  
  filepath <- paste0("images/", filename, ".jpg")
  
  # Download image
  try(
    download.file(url, filepath, mode = "wb", quiet = TRUE),
    silent = TRUE
  )
}

# ========================================================================
# 10. RUN IMAGE DOWNLOAD LOOP
# ========================================================================

cat("Starting image download...\n")

for (i in seq_len(nrow(df_img))) {
  cat("Downloading image", i, "of", nrow(df_img), "\r")
  download_image(i)
}

cat("\nDownload completed successfully!\n")

Pro Tips

Replace `key` with the taxonKey returned by name_backbone() for any taxon.

The name of the figure can be adjusted according to interest, in this example we use all these characters. sp, collector, rec_num, herb, cat_num, country, lat and lon.
